# Aseguramos la repetibilidad de los datos aleatorios
set.seed(2026)

# Definimos la cantidad total de archivos solicitados
total_archivos <- 40

# Ciclo para generar los 40 archivos de manera automatizada
for (i in 1:total_archivos) {
  
  # 1. Generamos los valores aleatorios uniformes
  # Generamos, por ejemplo, 30 datos aleatorios entre 0 y 100 por cada archivo
  valores_uniformes <- runif(30, min = 0, max = 100)
  
  # 2. Creamos los nombres dinámicos usando el índice del ciclo (i)
  nombre_txt <- paste0("archivo_", i, ".txt")
  nombre_pdf <- paste0("grafico_", i, ".pdf")
  
  # 3. Escritura del archivo de texto (.txt)
  # El argumento "\n" asegura que los datos se separen correctamente
  write(valores_uniformes, file = nombre_txt, sep = "\n")
  
  # 4. Generación del archivo gráfico (.pdf)
  pdf(nombre_pdf) # Abrimos el dispositivo PDF
  
  # Graficamos el vector usando tipo "l" (línea) para visualizar los datos uniformes
  plot(valores_uniformes, 
       type = "l", 
       col = "darkblue", 
       main = paste("Distribución Uniforme - Archivo", i),
       xlab = "Índice de Datos", 
       ylab = "Valor Aleatorio")
  
  dev.off() # Apagamos y cerramos el dispositivo PDF para guardar el archivo
}
